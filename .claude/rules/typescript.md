# TypeScript Rules

Applies to all `.ts` and `.tsx` files generated by any agent.

## Type Safety
- `strict: true` in `tsconfig.json` — non-negotiable
- No `any` — use `unknown` and narrow with type guards, or define a proper type
- No `as` type assertions unless you have a cast comment explaining why
- No `!` non-null assertions unless you've verified the value cannot be null
- Prefer `interface` over `type` for object shapes; use `type` for unions/intersections

## Null Handling
```typescript
// Bad
const name = user.name!

// Good
if (!user.name) throw new NotFoundError('User has no name')
const name = user.name
```

## Enums → Use String Literal Unions
```typescript
// Bad
enum Status { Active, Inactive }

// Good
type Status = 'active' | 'inactive'
const STATUS = { ACTIVE: 'active', INACTIVE: 'inactive' } as const
```

## Function Signatures
```typescript
// Always type parameters and return values on exported functions
export async function createUser(input: CreateUserInput): Promise<User> { ... }

// Use object params for 3+ arguments
function configure({ host, port, timeout }: ServerConfig): void { ... }
```

## Generics
```typescript
// Name generics descriptively — not just T
function fetchResource<TResource>(id: string): Promise<TResource> { ... }

// Constrain when possible
function getProperty<TObject, TKey extends keyof TObject>(obj: TObject, key: TKey) { ... }
```

## Imports
```typescript
// Type-only imports must use `import type`
import type { User } from './types'
import type { NextRequest } from 'next/server'

// Use path aliases — never deep relative imports
import { UserService } from '@/services/user'   // not: '../../../services/user'
```

## Zod for Runtime Validation
```typescript
import { z } from 'zod'

// Define schema first, derive type from it
const CreateUserSchema = z.object({
  email: z.string().email(),
  password: z.string().min(8),
  name: z.string().min(1).max(100),
})
type CreateUserInput = z.infer<typeof CreateUserSchema>

// Always use safeParse at boundaries
const result = CreateUserSchema.safeParse(requestBody)
if (!result.success) throw new ValidationError(result.error.flatten().fieldErrors)
```

## Async / Await
```typescript
// Always await Promises — never fire-and-forget without error handling
// Bad: doSomethingAsync()
// Good:
void doSomethingAsync().catch(logger.error)  // intentional fire-and-forget
await doSomethingAsync()                      // normal usage

// Handle Promise.all failures explicitly
const [a, b] = await Promise.all([fetchA(), fetchB()])
// Use Promise.allSettled when partial failure is acceptable
```

## React / TSX Specific
```typescript
// Component props: always explicit interface
interface ButtonProps {
  children: React.ReactNode
  onClick?: () => void
  disabled?: boolean
  variant?: 'primary' | 'secondary' | 'ghost'
}

// Event handlers: use React's event types
const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => { ... }
const handleSubmit = (e: React.FormEvent<HTMLFormElement>) => { ... }

// Keys in lists: stable, unique, never array index for dynamic lists
items.map((item) => <Item key={item.id} item={item} />)
```

## tsconfig.json Baseline
```json
{
  "compilerOptions": {
    "strict": true,
    "noUncheckedIndexedAccess": true,
    "exactOptionalPropertyTypes": true,
    "noImplicitReturns": true,
    "noFallthroughCasesInSwitch": true,
    "paths": { "@/*": ["./src/*"] }
  }
}
```
